# BUILD SEMANTIC CORPUS
# ============================================================================

def build_semantic_texts():
    """Build rich descriptive texts for semantic embedding."""
    texts, meta = [], []

    for s in SURAHS:
        text = (
            f"Surah {s['name_english']} is recommended for "
            f"{', '.join(s['categories'])}. "
            f"{s['themes']}. {s['keywords']}. "
            f"Note: {s.get('note', '')}."
        )
        texts.append(text)
        meta.append({"type": "surah", "data": s})

    for v in VERSES:
        text = (
            f"The verse {v['reference']} known as {v['name']} says: "
            f"{v['translation']} "
            f"It is recommended for {', '.join(v['categories'])}. "
            f"{v['keywords']}."
        )
        texts.append(text)
        meta.append({"type": "verse", "data": v})

    for d in DUAS:
        text = (
            f"The dua called {d['name']} says: "
            f"{d['translation']} "
            f"It is recommended for {', '.join(d['categories'])}. "
            f"{d['keywords']}."
        )
        texts.append(text)
        meta.append({"type": "dua", "data": d})

    return texts, meta

# ALSO BUILD TF-IDF FOR COMPARISON

def build_tfidf_corpus():
    corpus = []
    for s in SURAHS:
        corpus.append(s["themes"] + " " + s["keywords"] + " " +
                      " ".join(s["categories"]))
    for v in VERSES:
        corpus.append(v["keywords"] + " " + v["translation"] + " " +
                      " ".join(v["categories"]))
    for d in DUAS:
        corpus.append(d["keywords"] + " " + d["translation"] + " " +
                      " ".join(d["categories"]))
    return corpus

tfidf_corpus = build_tfidf_corpus()
vectorizer = TfidfVectorizer(ngram_range=(1,2), max_features=500,
                              stop_words="english")
tfidf_matrix = vectorizer.fit_transform(tfidf_corpus)

# ============================================================================
# SEMANTIC RECOMMENDATION FUNCTION
# ============================================================================
def semantic_recommend(query_text, top_n_surahs=3,
                       top_n_verses=3, top_n_duas=2):
    """
    Recommend using semantic embeddings.
    Understands meaning even without exact keyword matches.
    """
    query_embedding = model.encode([query_text], convert_to_numpy=True)
    scores = cosine_similarity(query_embedding, corpus_embeddings).flatten()

    surah_results = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "surah"]
    verse_results = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "verse"]
    dua_results   = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "dua"]

    surah_results.sort(key=lambda x: x[0], reverse=True)
    verse_results.sort(key=lambda x: x[0], reverse=True)
    dua_results.sort(key=lambda x: x[0], reverse=True)

    return {
        "surahs": surah_results[:top_n_surahs],
        "verses": verse_results[:top_n_verses],
        "duas":   dua_results[:top_n_duas]
    }

def tfidf_recommend(query_text, top_n_surahs=3,
                    top_n_verses=3, top_n_duas=2):
    """TF-IDF recommendation for comparison."""
    query_vec = vectorizer.transform([query_text])
    scores = cosine_similarity(query_vec, tfidf_matrix).flatten()

    surah_results = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "surah"]
    verse_results = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "verse"]
    dua_results   = [(scores[i], meta[i]["data"])
                     for i in range(len(meta))
                     if meta[i]["type"] == "dua"]

    surah_results.sort(key=lambda x: x[0], reverse=True)
    verse_results.sort(key=lambda x: x[0], reverse=True)
    dua_results.sort(key=lambda x: x[0], reverse=True)

    return {
        "surahs": surah_results[:top_n_surahs],
        "verses": verse_results[:top_n_verses],
        "duas":   dua_results[:top_n_duas]
    }

def format_results(situation, results):

        print(f"       {item['arabic']}")

    print("\nRELEVANT DUAS:")
    for score, item in results["duas"]:

        print(f"       {item['arabic']}")
        print(f"       {item['translation']}")
        print(f"       Source: {item['source']}")
